46  uint16_t num_inhibitory_inputs, const input_t *inh_input,
47  input_t external_bias, neuron_t *restrict neuron) {
48 
49 // log_debug("Exc 1: %12.6k, Exc 2: %12.6k", exc_input[0], exc_input[1]);
50 // log_debug("Inh 1: %12.6k, Inh 2: %12.6k", inh_input[0], inh_input[1]);
51 
52  // If outside of the refractory period
53  if (neuron->refract_timer <= 0) {
54  REAL total_exc = 0;
55  REAL total_inh = 0;
56 
57  for (int i=0; i < num_excitatory_inputs; i++) {
58  total_exc += exc_input[i];
59  log_info("Exc %d: %12.6k\ttotal: %k", i, exc_input[i], total_exc);
60  }
61  for (int i=0; i< num_inhibitory_inputs; i++) {
62  total_inh += inh_input[i];
63 // log_info("Inh %d: %12.6k", i, inh_input[i]);
64  }
65  // Get the input in nA
66  input_t input_this_timestep =
67  total_exc - total_inh + external_bias + global_params->I_offset;
68  log_info("total_input %k", input_this_timestep);
69  lif_neuron_closed_form(
70  neuron, neuron->V_membrane, input_this_timestep);
71  } else {
72  // countdown refractory timer
73  neuron->refract_timer--;
74  }
75  return neuron->V_membrane;
76 }
